CREATE TABLE IF NOT EXISTS tip(
business_id STRING,
    user_id STRING,
       text STRING,
         ds DATE,
      likes INT
) ROW FORMAT DELIMITED FIELDS TERMINATED BY '\t';

CREATE TABLE IF NOT EXISTS review(
business_id STRING,
    user_id STRING,
      stars DECIMAL(10,6),
       text STRING,
         ds DATE,
      votes INT
) ROW FORMAT DELIMITED FIELDS TERMINATED BY '\t';

CREATE TABLE IF NOT EXISTS business(
  business_id STRING,
         name STRING,
 full_address STRING,
         city STRING,
        state STRING,
         open BOOLEAN,
   categories STRING,
 review_count INT,
        stars DECIMAL(10,6),
neighborhoods STRING,
    longitude DECIMAL(10,6),
     latitude DECIMAL(10,6),
   attributes STRING
) ROW FORMAT DELIMITED FIELDS TERMINATED BY '\t';

CREATE TABLE IF NOT EXISTS users(
          user_id STRING,
             name STRING,
     review_count INT,
    average_stars DECIMAL(10,6),
            votes STRING,
          friends STRING,
            elite STRING,
    yelping_since STRING,
      compliments STRING,
             fans INT
) ROW FORMAT DELIMITED FIELDS TERMINATED BY '\t';
